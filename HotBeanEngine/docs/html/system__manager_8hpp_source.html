140 void EntitySignatureChanged(Entity entity, Signature entitySignature)

142 // Notify each system that an entity's signature changed

144 auto const& systemSignature = _signatures[typeName];
145
146 // Entity signature matches system signature - insert into set

148 system->_entities.insert(entity);

150 // Entity signature does not match system signature - erase from set
