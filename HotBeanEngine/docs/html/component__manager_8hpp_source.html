113 template<typename T>
114 ComponentType RemoveComponent(Entity entity) {
115 static_assert(std::is_base_of_v<Component, T>, "T must inherit from Component!");
116
117 T component;
118 std::string component_name = component.GetName();
119 assert(!component_name.empty() && "Component name not set!");
120 assert(m_component_name_to_type.find(component_name) != m_component_name_to_type.end() && "Component not registered!");
121
122 if (!m_testing) {
123 m_logging_manager->Log(LoggingType::DEBUG, "Removing Component \"" + component_name + "\" from"
124 " Entity \"" + std::to_string(entity) + "\"");
125 }
126
127 std::shared_ptr<SparseSet<T, MAX_ENTITIES>> sparseSet = GetComponentSet<T>();
128
129 // Removes entity from component sparse set
130 sparseSet->Remove(entity);
131
132 // If the last component is removed, the component array must be destroyed and unregistered
133 if (sparseSet->Size() == 0) {
134 if (!m_testing)
135 m_logging_manager->Log(LoggingType::DEBUG, "\tAll \"" + component_name + "\" Components removed... Destroying Component Array");
136
137 m_component_name_to_type.erase(component_name);
138 m_component_name_to_data.erase(component_name);
139 m_registered_components--;
140 }
141
142 // Return component type to update entity signature
143 return m_component_name_to_type[component_name];
144 }
