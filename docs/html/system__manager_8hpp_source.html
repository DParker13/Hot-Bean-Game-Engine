130 void EntitySignatureChanged(Entity entity, Signature entitySignature)
131 {
132 // Notify each system that an entity's signature changed
133 for (auto& [typeName, system] : _systems) {
134 auto const& systemSignature = _signatures[typeName];
135
136 // Entity signature matches system signature - insert into set
137 if ((entitySignature & systemSignature) == systemSignature) {
138 system->_entities.insert(entity);
139 }
140 // Entity signature does not match system signature - erase from set
141 else {
142 system->_entities.erase(entity);
