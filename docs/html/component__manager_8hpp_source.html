62 When you add a component to an entity, check if the component is registered.

93 template<typename T>
94 ComponentType RemoveComponent(Entity entity) {
95 // Component Type Name
96 std::string typeName = typeid(T).name();
97
98 assert(_componentNameToType.find(typeName) != _componentNameToType.end() && "Component not registered!");
99
100 std::shared_ptr<SparseSet<T, MAX_ENTITIES>> sparseSet = GetComponentSet<T>();
101
102 // Removes entity from component sparse set
103 sparseSet->Remove(entity);
104
105 // If the last component is removed, the component array must be destroyed and unregistered
106 if (sparseSet->Size() == 0) {
107 _componentNameToType.erase(typeName);
108 _componentNameToData.erase(typeName);
109 _registeredComponents--;
110 }
111
112 // Return component type to update entity signature
113 return _componentNameToType[typeName];
114 }
